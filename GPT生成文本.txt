# OpenAI Provider

```javascript
class OpenAIProvider 
extends AIProvider{


async chat(message){


let response =
await fetch(

"https://api.openai.com/v1/chat/completions",

{


method:"POST",


headers:{


"Content-Type":
"application/json",


"Authorization":
"Bearer "+
this.config.apiKey


},



body:JSON.stringify({


model:
this.config.model
||
"gpt-4.1-mini",



messages:[

{

role:"user",

content:message

}

]


})


}

);



let data =
await response.json();


return (

data
.choices
?.[0]
?.message
?.content

||
""

);


}


}
```
